#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
#include<numeric>

using namespace std;

int main(int argc, char** argv)
{
	int test_case;
	int T;	
	vector<int> arr;
	scanf(" %d", &T);
	int K;
	int temp;
	for(test_case = 1; test_case <= T; ++test_case)
	{
		scanf(" %d", &K);
		for (int j = 0; j < K; j++) {
			scanf(" %d", &temp);
			if (temp > 0) {
				arr.push_back(temp);
			}
			else {
				if (arr.size() >0) {
					arr.pop_back();
				}
			}
		}

		int result = accumulate(arr.begin(), arr.end(), 0);
		printf("#%d %d\n", test_case, result);
		arr.clear();


	}
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
